<?php
require 'includes/auth_check.php';
// Solo admin puede gestionar usuarios
if ($_SESSION['rol'] !== 'admin') {
    header("Location: dashboard.php");
    exit();
}
require 'includes/config.php';

$error = $success = '';

// ----------------------------
// CONTAR ADMINISTRADORES
// ----------------------------
$stmt = $pdo->query("SELECT COUNT(*) FROM usuarios WHERE rol = 'admin'");
$total_admins = (int)$stmt->fetchColumn();

// ----------------------------
// PROCESAR FORMULARIO (CREAR/EDITAR)
// ----------------------------
if ($_SERVER['REQUEST_METHOD'] === 'POST') {
    try {
        $id = $_POST['id'] ?? null;
        $nombre = trim($_POST['nombre'] ?? '');
        $email = trim($_POST['email'] ?? '');
        $rol = $_POST['rol'] ?? 'vendedor';
        $password = $_POST['password'] ?? '';

        // Validaciones básicas
        if (empty($nombre) || empty($email)) {
            throw new Exception("Nombre y correo son obligatorios.");
        }
        if (!filter_var($email, FILTER_VALIDATE_EMAIL)) {
            throw new Exception("Correo electrónico no válido.");
        }

        // Verificar unicidad del email
        $checkSql = "SELECT id FROM usuarios WHERE email = ?";
        $params = [$email];
        if ($id) $checkSql .= " AND id != ?";
        if ($id) $params[] = $id;

        $check = $pdo->prepare($checkSql);
        $check->execute($params);
        if ($check->fetch()) {
            throw new Exception("Ya existe un usuario con ese correo.");
        }

        // Protección: no permitir cambios peligrosos
        if ($id) {
            // Si es el usuario actual, no permitir cambios
            if ($id == $_SESSION['usuario_id']) {
                throw new Exception("No puedes modificar tu propia cuenta desde aquí.");
            }

            // Si es el último admin, no permitir cambiar su rol
            if ($total_admins === 1) {
                $stmt = $pdo->prepare("SELECT rol FROM usuarios WHERE id = ?");
                $stmt->execute([$id]);
                $rol_actual = $stmt->fetchColumn();
                if ($rol_actual === 'admin' && $rol !== 'admin') {
                    throw new Exception("No puedes cambiar el rol del último administrador.");
                }
            }

            // Actualizar sin contraseña (solo si se proporciona)
            if (!empty($password)) {
                $hashed_password = password_hash($password, PASSWORD_DEFAULT);
                $stmt = $pdo->prepare("UPDATE usuarios SET nombre = ?, email = ?, rol = ?, password = ? WHERE id = ?");
                $stmt->execute([$nombre, $email, $rol, $hashed_password, $id]);
            } else {
                $stmt = $pdo->prepare("UPDATE usuarios SET nombre = ?, email = ?, rol = ? WHERE id = ?");
                $stmt->execute([$nombre, $email, $rol, $id]);
            }
            $success = "Usuario actualizado correctamente.";
        } else {
            // Crear nuevo usuario
            if (empty($password)) {
                throw new Exception("La contraseña es obligatoria al crear un usuario.");
            }
            $hashed_password = password_hash($password, PASSWORD_DEFAULT);
            $stmt = $pdo->prepare("INSERT INTO usuarios (nombre, email, password, rol) VALUES (?, ?, ?, ?)");
            $stmt->execute([$nombre, $email, $hashed_password, $rol]);
            $success = "Usuario creado correctamente.";
        }

    } catch (Exception $e) {
        $error = "Error: " . $e->getMessage();
    }
}

// ----------------------------
// ELIMINAR usuario
// ----------------------------
if (isset($_GET['eliminar'])) {
    $id = (int)$_GET['eliminar'];

    // Protecciones
    if ($id == $_SESSION['usuario_id']) {
        $error = "No puedes eliminarte a ti mismo.";
    } elseif ($total_admins === 1) {
        $stmt = $pdo->prepare("SELECT rol FROM usuarios WHERE id = ?");
        $stmt->execute([$id]);
        if ($stmt->fetchColumn() === 'admin') {
            $error = "No puedes eliminar al único administrador.";
        }
    } else {
        $pdo->prepare("DELETE FROM usuarios WHERE id = ?")->execute([$id]);
        $success = "Usuario eliminado correctamente.";
        header("Location: usuarios.php?eliminado=1");
        exit();
    }
}

// ----------------------------
// Cargar usuario para edición
// ----------------------------
$usuario_edit = null;
if (isset($_GET['editar'])) {
    if ((int)$_GET['editar'] == $_SESSION['usuario_id']) {
        $error = "No puedes editar tu propia cuenta desde aquí.";
    } else {
        $stmt = $pdo->prepare("SELECT id, nombre, email, rol FROM usuarios WHERE id = ?");
        $stmt->execute([(int)$_GET['editar']]);
        $usuario_edit = $stmt->fetch(PDO::FETCH_ASSOC);
    }
}

// ----------------------------
