<?php
session_start();
if (isset($_SESSION['usuario_id'])) {
    header("Location: dashboard.php");
    exit();
}

$error = '';
$empresa = null;

/* 🔹 Obtener datos de la empresa (logo + nombre) */
try {
    $pdo_empresa = new PDO(
        "mysql:host=localhost;dbname=pos_sistema;charset=utf8mb4",
        'root',
        '',
        [PDO::ATTR_ERRMODE => PDO::ERRMODE_EXCEPTION]
    );

    $stmtEmpresa = $pdo_empresa->query("SELECT nombre, logo_url FROM empresa LIMIT 1");
    $empresa = $stmtEmpresa->fetch(PDO::FETCH_ASSOC);

} catch (Exception $e) {
    $empresa = null;
}

/* 🔹 Login */
if ($_SERVER['REQUEST_METHOD'] === 'POST') {
    $email = $_POST['email'] ?? '';
    $password = $_POST['password'] ?? '';

    if (!$email || !$password) {
        $error = "Todos los campos son obligatorios.";
    } else {
        try {
            $stmt = $pdo_empresa->prepare("SELECT * FROM usuarios WHERE email = ?");
            $stmt->execute([$email]);
            $usuario = $stmt->fetch(PDO::FETCH_ASSOC);

            if ($usuario && password_verify($password, $usuario['password'])) {
                $_SESSION['usuario_id'] = $usuario['id'];
                $_SESSION['nombre']     = $usuario['nombre'];
                $_SESSION['rol']        = $usuario['rol'];
                $_SESSION['email']      = $usuario['email'];
                $_SESSION['empresa']    = $empresa;

                header("Location: dashboard.php");
                exit();
            } else {
                $error = "Credenciales incorrectas.";
            }
        } catch (Exception $e) {
            $error = "Error interno del sistema.";
        }
    }
}
