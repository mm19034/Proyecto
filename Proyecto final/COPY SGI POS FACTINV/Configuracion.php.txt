// ----------------------------
// PROCESAR ACTUALIZACIÓN
// ----------------------------
if ($_SERVER['REQUEST_METHOD'] === 'POST') {
    try {
        $nombre = trim($_POST['nombre'] ?? '');
        $nit = trim($_POST['nit'] ?? '');
        $direccion = trim($_POST['direccion'] ?? '');
        $telefono = trim($_POST['telefono'] ?? '');
        $email = trim($_POST['email_empresa'] ?? '');
        $iva = str_replace(',', '.', $_POST['iva_porcentaje'] ?? '13');
        $moneda = $_POST['moneda'] ?? 'USD';
        $stock_minimo = max(0, (int)($_POST['stock_minimo'] ?? 5));
        $prefijo_factura = trim($_POST['prefijo_factura'] ?? 'FAC');
        $ultimo_numero_factura = max(1, (int)($_POST['ultimo_numero_factura'] ?? 1));

        if (empty($nombre)) {
            throw new Exception("El nombre de la empresa es obligatorio.");
        }

        $iva = (float)$iva;
        if ($iva < 0 || $iva > 100) {
            throw new Exception("El IVA debe estar entre 0 y 100.");
        }

        // Subir logo
        $logo_url = $empresa['logo_url'];
        if (!empty($_FILES['logo']['name'])) {
            $upload_dir = 'uploads/empresa/';
            if (!is_dir($upload_dir)) mkdir($upload_dir, 0777, true);

            $file_name = $_FILES['logo']['name'];
            $file_tmp = $_FILES['logo']['tmp_name'];
            $file_ext = strtolower(pathinfo($file_name, PATHINFO_EXTENSION));
            $allowed_ext = ['jpg', 'jpeg', 'png', 'webp', 'gif'];

            if (in_array($file_ext, $allowed_ext)) {
                if ($logo_url && file_exists($logo_url)) {
                    unlink($logo_url);
                }
                $filename = 'logo.' . $file_ext;
                if (move_uploaded_file($file_tmp, $upload_dir . $filename)) {
                    $logo_url = $upload_dir . $filename;
                }
            } else {
                throw new Exception("Formato de imagen no permitido. Use JPG, PNG, WEBP o GIF.");
            }
        }

        // Actualizar en BD
        $stmt = $pdo->prepare("
            UPDATE empresa 
            SET nombre = ?, nit = ?, direccion = ?, telefono = ?, 
                email = ?, logo_url = ?, iva_porcentaje = ?, moneda = ?, 
                stock_minimo = ?, prefijo_factura = ?, ultimo_numero_factura = ?
            WHERE id = ?
        ");
        $stmt->execute([
            $nombre, $nit, $direccion, $telefono, $email, $logo_url,
            $iva, $moneda, $stock_minimo, $prefijo_factura, $ultimo_numero_factura,
            $empresa['id']
        ]);
        
        // Recargar datos
        $empresa = $pdo->query("SELECT * FROM empresa LIMIT 1")->fetch(PDO::FETCH_ASSOC);
        $_SESSION['empresa'] = $empresa;
        $success = "Configuración actualizada correctamente.";

    } catch (Exception $e) {
        $error = "Error: " . $e->getMessage();
    }
}

    document.getElementById('logoInput').addEventListener('change', function(e) {
        const preview = document.querySelector('.logo-preview');
        const file = e.target.files[0];
        if (file) {
            const reader = new FileReader();
            reader.onload = function(event) {
                preview.innerHTML = `<img src="${event.target.result}" alt="Vista previa">`;
            };
            reader.readAsDataURL(file);
        }
    });
