if (isset($_POST['agregar_producto'])) {
    $producto_id = (int)$_POST['producto_id'];
    $cantidad = (int)$_POST['cantidad'];

    if ($producto_id && $cantidad > 0) {
        $stmt = $pdo->prepare("
            SELECT p.id, p.nombre, p.precio_venta, p.stock,
                   d.tipo as descuento_tipo, d.valor as descuento_valor
            FROM productos p
            LEFT JOIN descuentos d ON p.id = d.producto_id 
                AND d.activo = 1 
                AND (d.fecha_fin IS NULL OR d.fecha_fin >= CURDATE())
            WHERE p.id = ? AND p.estado = 'activo'
        ");
        $stmt->execute([$producto_id]);
        $producto = $stmt->fetch(PDO::FETCH_ASSOC);

        if ($producto && $cantidad <= $producto['stock']) {
            $existe = false;
            foreach ($_SESSION['carrito'] as &$item) {
                if ($item['id'] == $producto_id) {
                    $item['cantidad'] += $cantidad;
                    if ($item['cantidad'] > $producto['stock']) {
                        $item['cantidad'] = $producto['stock'];
                    }
                    $existe = true;
                    break;
                }
            }
            if (!$existe) {
                $descuento_aplicado = 0;
                if ($producto['descuento_tipo']) {
                    if ($producto['descuento_tipo'] === 'porcentaje') {
                        $descuento_aplicado = $producto['precio_venta'] * ($producto['descuento_valor'] / 100);
                    } else {
                        $descuento_aplicado = $producto['descuento_valor'];
                    }
                    $descuento_aplicado = min($descuento_aplicado, $producto['precio_venta']);
                }

                $_SESSION['carrito'][] = [
                    'id' => $producto_id,
                    'nombre' => $producto['nombre'],
                    'precio' => (float)$producto['precio_venta'],
                    'cantidad' => $cantidad,
                    'descuento' => (float)$descuento_aplicado
                ];
            }
            header("Location: ventas.php");
            exit();
        } else {
            $error = "Producto no disponible o stock insuficiente.";
        }
    }

            $nuevo_cliente_email = trim($_POST['nuevo_cliente_email'] ?? '');

            $cliente_nombre = 'Cliente genérico';
            if ($cliente_id) {
                $stmt = $pdo->prepare("SELECT nombre FROM clientes WHERE id = ?");
                $stmt->execute([$cliente_id]);
                $cliente_row = $stmt->fetch();
                if ($cliente_row) {
                    $cliente_nombre = $cliente_row['nombre'];

            $referencia = trim($_POST['referencia'] ?? '');

            $subtotal = 0;
            $descuento_total = 0;
            foreach ($_SESSION['carrito'] as $item) {
                $item_total = $item['precio'] * $item['cantidad'];
                $descuento_item = $item['descuento'] * $item['cantidad'];
                $subtotal += $item_total;
                $descuento_total += $descuento_item;
            }
            $subtotal_con_descuento = $subtotal - $descuento_total;
            $iva = $subtotal_con_descuento * ($iva_porcentaje / 100);
            $total = $subtotal_con_descuento + $iva;

            $fecha_venta = date('Y-m-d H:i:s');
            $codigo_venta = 'VTA-' . date('Ymd') . '-' . str_pad(mt_rand(1, 9999), 4, '0', STR_PAD_LEFT);

            // ✅ Generar número de factura correlativo
            $prefijo = $empresa['prefijo_factura'] ?? 'FAC';

            $numero_factura = $prefijo . '-' . str_pad($nuevo_numero, 6, '0', STR_PAD_LEFT);

            $stmt = $pdo->prepare("
                INSERT INTO ventas (
                    cliente_id, usuario_id, codigo_venta, numero_factura, subtotal, iva_aplicado, 
                    descuento_total, total, metodo_pago, referencia_pago, estado, fecha
                ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, 'completada', ?)
            ");
            $stmt->execute([
                $cliente_id, 
                $_SESSION['usuario_id'], 
                $codigo_venta,
                $numero_factura,
                $subtotal, 
                $iva, 
                $descuento_total, 
                $total,
                $metodo_pago, 
                $referencia,
                $fecha_venta
            ]);
            $venta_id = $pdo->lastInsertId();

            // ✅ Actualizar último número de factura
            $pdo->prepare("UPDATE empresa SET ultimo_numero_factura = ? WHERE id = ?")
                 ->execute([$nuevo_numero + 1, $empresa['id']]);

            foreach ($_SESSION['carrito'] as $item) {
                $stmt = $pdo->prepare("
                    INSERT INTO detalle_ventas (
                        venta_id, producto_id, cantidad, precio_unitario, 
                        descuento_aplicado, subtotal
                    ) VALUES (?, ?, ?, ?, ?, ?)
                ");
                $subtotal_item = ($item['precio'] - $item['descuento']) * $item['cantidad'];
                $stmt->execute([
                    $venta_id, $item['id'], $item['cantidad'],
                    $item['precio'], $item['descuento'], $subtotal_item
                ]);

                $pdo->prepare("UPDATE productos SET stock = stock - ? WHERE id = ?")
                     ->execute([$item['cantidad'], $item['id']]);
            }

            // ===== GENERAR PDF =====
            $pdf_content = generarFacturaPDF(
                $venta_id,
                $numero_factura,
                $codigo_venta,
                $fecha_venta,
                $total,
                $_SESSION['carrito'],
                $cliente_nombre,
                $metodo_pago,
                $iva,
                $iva_porcentaje,
                $empresa
            );

            // ===== ENVIAR POR CORREO =====
            $email_cliente = null;
            if ($cliente_id) {
                $stmt = $pdo->prepare("SELECT email FROM clientes WHERE id = ?");
                $stmt->execute([$cliente_id]);
                $cliente_row = $stmt->fetch();
                $email_cliente = $cliente_row ? $cliente_row['email'] : null;
            } elseif ($nuevo_cliente_email) {
                $email_cliente = $nuevo_cliente_email;
            }

            if ($email_cliente) {
                $mail = new PHPMailer(true);
                try {
                    // Configuración SMTP (ejemplo con Gmail)
                    $mail->isSMTP();
                    $mail->Host       = 'smtp.gmail.com';
                    $mail->SMTPAuth   = true;
                    $mail->Username   = 'sgipos.sistema@gmail.com';   // Tu correo
                    $mail->Password   = 'yygk zazu jjzf vgez';     // Contraseña de app
                    $mail->SMTPSecure = PHPMailer::ENCRYPTION_STARTTLS;
                    $mail->Port       = 587;

                    $mail->setFrom('sgipos.sistema@gmail.com', 'SGI-POS');
                    $mail->addAddress($email_cliente, $cliente_nombre);

                    $mail->isHTML(true);
                    $mail->CharSet = 'UTF-8';
                    $mail->Subject = "Factura Electrónica - {$numero_factura}";
                    $mail->Body    = "
                        <p>Hola {$cliente_nombre},</p>
                        <p>Adjuntamos su factura electrónica.</p>
                        <p>Gracias por su compra.</p>
                        <p>Saludos,</p>
                        <p>Equipo SGI-POS</p>
                    ";
                    $mail->addStringAttachment($pdf_content, "factura_{$numero_factura}.pdf", 'base64', 'application/pdf');

                    $mail->send();
                    $success .= " Factura enviada a {$email_cliente}.";
                } catch (Exception $e) {
                    error_log("Error al enviar factura: " . $mail->ErrorInfo);
                    $error = "No se pudo enviar la factura por correo.";
                }
            }

            // ===== GUARDAR PDF EN ARCHIVO =====
            $nombre_pdf = "ticket_{$codigo_venta}.pdf";
            file_put_contents("tickets/{$nombre_pdf}", $pdf_content);

            // ===== GUARDAR EN SESIÓN =====
            $_SESSION['ticket_venta'] = [
                'numero_factura' => $numero_factura,
                'codigo' => $codigo_venta,
                'fecha' => $fecha_venta,
                'total' => $total,
                'items' => $_SESSION['carrito'],
                'cliente_nombre' => $cliente_nombre,
                'metodo_pago' => $metodo_pago,
                'iva' => $iva,
                'iva_porcentaje' => $iva_porcentaje,
                'empresa' => $empresa,
                'archivo' => $nombre_pdf
            ];

            unset($_SESSION['carrito']);
            $pdo->commit();
            header("Location: ventas.php?ticket=1");
            exit();

        } catch (Exception $e) {
            $pdo->rollback();
            $error = "Error al procesar la venta: " . $e->getMessage();
        }
    }

<script src="https://cdn.jsdelivr.net/npm/bootstrap@5.3.3/dist/js/bootstrap.bundle.min.js"></script>
<script>
    // Mostrar modal al hacer clic en producto
    document.querySelectorAll('.product-card').forEach(card => {
        card.addEventListener('click', () => {
            const id = card.dataset.id;
            const nombre = card.dataset.nombre;
            document.getElementById('productoId').value = id;
            document.getElementById('modalTitulo').textContent = 'Agregar: ' + nombre;
            const modal = new bootstrap.Modal(document.getElementById('modalAgregar'));
            modal.show();
        });
    });

    // Búsqueda de productos
    document.getElementById('btnBuscar').addEventListener('click', buscarProductos);
    document.getElementById('buscarProducto').addEventListener('keypress', (e) => {
        if (e.key === 'Enter') buscarProductos();
    });

    function buscarProductos() {
        const query = document.getElementById('buscarProducto').value.toLowerCase();
        document.querySelectorAll('.product-card').forEach(card => {
            const nombre = card.dataset.nombre.toLowerCase();
            card.style.display = nombre.includes(query) ? 'block' : 'none';
        });
    }

    // Mostrar campo de referencia según método de pago
    document.getElementById('metodoPago').addEventListener('change', function() {
        const campo = document.getElementById('campoReferencia');
        campo.style.display = (this.value !== 'efectivo') ? 'block' : 'none';
    });

    // Formulario de procesar venta
    document.getElementById('btnProcesar').addEventListener('click', function() {
        const form = document.createElement('form');
        form.method = 'POST';

        const clienteId = document.querySelector('[name="cliente_id"]').value;
        const nuevoNombre = document.querySelector('[name="nuevo_cliente_nombre"]').value;
        const nuevoEmail = document.querySelector('[name="nuevo_cliente_email"]').value;
        const metodo = document.getElementById('metodoPago').value;
        const referencia = document.querySelector('[name="referencia"]')?.value || '';

        const clienteIdInput = document.createElement('input');
        clienteIdInput.type = 'hidden';
        clienteIdInput.name = 'cliente_id';
        clienteIdInput.value = clienteId;
        form.appendChild(clienteIdInput);

        if (nuevoNombre) {
            const nuevoNombreInput = document.createElement('input');
            nuevoNombreInput.type = 'hidden';
            nuevoNombreInput.name = 'nuevo_cliente_nombre';
            nuevoNombreInput.value = nuevoNombre;
            form.appendChild(nuevoNombreInput);
        }

        if (nuevoEmail) {
            const nuevoEmailInput = document.createElement('input');
            nuevoEmailInput.type = 'hidden';
            nuevoEmailInput.name = 'nuevo_cliente_email';
            nuevoEmailInput.value = nuevoEmail;
            form.appendChild(nuevoEmailInput);
        }

        const metodoInput = document.createElement('input');
        metodoInput.type = 'hidden';
        metodoInput.name = 'metodo_pago';
        metodoInput.value = metodo;
        form.appendChild(metodoInput);

        if (metodo !== 'efectivo' && referencia) {
            const refInput = document.createElement('input');
            refInput.type = 'hidden';
            refInput.name = 'referencia';
            refInput.value = referencia;
            form.appendChild(refInput);
        }

        const btnInput = document.createElement('input');
        btnInput.type = 'hidden';
        btnInput.name = 'procesar_venta';
        btnInput.value = '1';
        form.appendChild(btnInput);

        document.body.appendChild(form);
        form.submit();
    });

    // Ver detalle de venta
    function verDetalleVenta(ventaId) {
        fetch(`ventas.php?detalle=${ventaId}&ajax=1`)
            .then(response => response.text())
            .then(html => {
                document.getElementById('detalleVentaContent').innerHTML = html;
                const modal = new bootstrap.Modal(document.getElementById('modalDetalleVenta'));
                modal.show();
            })
            .catch(err => {
                document.getElementById('detalleVentaContent').innerHTML = '<div class="alert alert-danger">Error al cargar el detalle.</div>';
            });
    }
</script>
