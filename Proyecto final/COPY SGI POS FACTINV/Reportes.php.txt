// Función para obtener rango de fechas
function getRangoFechas($periodo) {
    $hoy = date('Y-m-d');
    switch ($periodo) {
        case 'hoy':
            return [$hoy, $hoy];
        case 'semana':
            $fecha_inicio = date('Y-m-d', strtotime('monday this week'));
            return [$fecha_inicio, $hoy];
        case 'mes':
            $fecha_inicio = date('Y-m-01');
            return [$fecha_inicio, $hoy];
        default:
            return [date('Y-m-d', strtotime('-30 days')), $hoy];
    }
}

// Determinar rango de fechas
if (isset($_GET['periodo'])) {
    $periodo = $_GET['periodo'];
    [$fecha_inicio, $fecha_fin] = getRangoFechas($periodo);
} elseif (isset($_GET['fecha_inicio']) && isset($_GET['fecha_fin'])) {
    $fecha_inicio = $_GET['fecha_inicio'];
    $fecha_fin = $_GET['fecha_fin'];
    $periodo = 'personalizado';
} else {
    $periodo = 'mes';
    [$fecha_inicio, $fecha_fin] = getRangoFechas($periodo);
}

// ✅ Convertir a fechas completas (con hora) para comparar con TIMESTAMP
$fecha_inicio_completa = $fecha_inicio . ' 00:00:00';
$fecha_fin_completa = $fecha_fin . ' 23:59:59';

// ✅ Usar comillas simples (sin PDO::quote)
$fecha_inicio_sql = "'$fecha_inicio_completa'";
$fecha_fin_sql = "'$fecha_fin_completa'";

// ----------------------------
// RESUMEN GENERAL
// ----------------------------
$stmt = $pdo->prepare("
    SELECT 
        COUNT(*) as total_ventas,
        COALESCE(SUM(total), 0) as total_ingresos,
        COALESCE(SUM(iva_aplicado), 0) as total_iva
    FROM ventas 
    WHERE fecha BETWEEN $fecha_inicio_sql AND $fecha_fin_sql
");
$stmt->execute();
$resumen = $stmt->fetch(PDO::FETCH_ASSOC);
if (!$resumen) {
    $resumen = ['total_ventas' => 0, 'total_ingresos' => 0, 'total_iva' => 0];
}

// ----------------------------
// PRODUCTOS MÁS VENDIDOS
// ----------------------------
$productos_top = $pdo->query("
    SELECT 
        p.nombre,
        SUM(dv.cantidad) as unidades_vendidas,
        COALESCE(SUM(dv.subtotal), 0) as ingresos
    FROM detalle_ventas dv
    JOIN productos p ON dv.producto_id = p.id
    JOIN ventas v ON dv.venta_id = v.id
    WHERE v.fecha BETWEEN $fecha_inicio_sql AND $fecha_fin_sql
    GROUP BY p.id, p.nombre
    ORDER BY unidades_vendidas DESC
    LIMIT 10
")->fetchAll(PDO::FETCH_ASSOC);

// ----------------------------
// CLIENTES FRECUENTES
// ----------------------------
$clientes_top = $pdo->query("
    SELECT 
        c.nombre,
        COUNT(v.id) as compras,
        COALESCE(SUM(v.total), 0) as total_gastado
    FROM ventas v
    JOIN clientes c ON v.cliente_id = c.id
    WHERE v.fecha BETWEEN $fecha_inicio_sql AND $fecha_fin_sql
    GROUP BY c.id, c.nombre
    ORDER BY compras DESC
    LIMIT 10
")->fetchAll(PDO::FETCH_ASSOC);

// Si no hay clientes, buscar ventas sin cliente
if (empty($clientes_top)) {
    $clientes_top = $pdo->query("
        SELECT 
            'Cliente no registrado' as nombre,
            COUNT(*) as compras,
            COALESCE(SUM(total), 0) as total_gastado
        FROM ventas 
        WHERE cliente_id IS NULL 
        AND fecha BETWEEN $fecha_inicio_sql AND $fecha_fin_sql
    ")->fetchAll(PDO::FETCH_ASSOC);
}
